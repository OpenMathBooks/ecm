\begin{align*} 1^2 + 2^2 + 3^2 + \cdots + n^2 = \amp \binom{n+1}{3} + \binom{n+2}{3} \\ 1^2 + 2^2 + \cdots +(n-1)^2 = \amp \binom{n}{3} + \binom{n+1}{3}\\ 1^2 + 2^2 + \cdots +(n-2)^2 = \amp \binom{n-1}{3} + \binom{n}{3}\\ \vdots \amp \vdots \\ 1^2 + 2^2 = \amp \binom{3}{3} + \binom{4}{3} \\ 1^2 = \amp \binom{3}{3} \end{align*}
